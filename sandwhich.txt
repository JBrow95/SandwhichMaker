#Lets Make A Sandwhich
def bread
  live_loop :beat do
    use_bpm 75
    
    with_fx :reverb, mix: 0.5 do
      sample :vinyl_hiss, amp: 3
    end
    sleep 0.5
  end
end

def veggies
  notes = sample :loop_mehackit2, rate: -0.5
  
  puts notes
  sleep 1
end

def meat
  sample :loop_mika, amp: 3
end

def mayo
  use_synth :fm
  
  play 69
  sleep 0.5
  play 62
  sleep 0.5
  play 58
  sleep 3
  play 58
  sleep 0.5
  play 60
  sleep 0.5
  play 53
  sleep 1
end

live_loop :chef do
  sync :tick
  bread
  veggies
  meat
  mayo
end

live_loop :chefcut do
  sync :tick
  bread
  veggies
  meat
  mayo
end

live_loop :tick do
  sleep 1
end
