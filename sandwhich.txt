#Lets Make A Sandwhich
def bread
  live_loop :beat do
    use_bpm 50
    
    with_fx :reverb, mix: 0.5 do
      sample :vinyl_hiss, amp: 3
    end
    sleep 0.5
  end
end

def veggies
  notes = sample :loop_mehackit2, rate: -0.5
  
  puts notes
  sleep 1
end

def meat
  sample :perc_snap
  sleep 1
end

def mayo
  use_synth :fm
  play 69
  sleep 0.5
  play 62
  sleep 0.5
  play 58
  sleep 2
  play 58
  sleep 0.5
  play 60
  sleep 0.5
  play 53
  sleep 1
end

live_loop :chef do
  bread
  veggies
  meat
  mayo
end
